% Math environments
\newenvironment{eqarrows}{
	\begin{equation}
		\begin{WithArrows}
		}{%
		\end{WithArrows}
	\end{equation}
	\ignorespacesafterend
}
\newenvironment{eqarrows*}[1][]{
	\begin{equation*}
		\begin{WithArrows}[#1]
		}{%
		\end{WithArrows}
	\end{equation*}
	\ignorespacesafterend
}
\newenvironment{eqboxed}{
	% https://tex.stackexchange.com/questions/419132/newenvironment-with-empheq-inside-tcolorbox
	\empheq[box=\fbox]{align}
	}{%
	\endempheq%
	\ignorespacesafterend
}
\newenvironment{eqboxed*}{
	% https://tex.stackexchange.com/questions/419132/newenvironment-with-empheq-inside-tcolorbox
	\empheq[box=\fbox]{align*}
	}{%
	\endempheq%
	\ignorespacesafterend
}
